function shuffle(array) {
    let shuffleArray = [];
    let max = array.length - 1;
    let min = 0;

    for (let i = 0; i < array.length; i++) {
        let j = 0;
        do {
            j = Math.round(min - 0.5 + Math.random() * (max - min + 0.5));
        } while (shuffleArray[j] != undefined);

        shuffleArray[j] = array[i];
    }
    return shuffleArray;
}
